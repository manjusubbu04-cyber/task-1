'timescale 1ns/1ps

module  demux4to1_tb;

reg I,S0,S1;
wire Y0, Y1, Y2, Y3;

demux1to4_behavioral uut(
    .S0(S0),
    .S1(S1),
    .I(I),
    .Y0(Y0),
    .Y1(Y1),
    .Y2(Y2),
    .Y3(Y3)
);

initial
begin

    $monitor( I=%b S0=%b S1=%b Y0=%b Y1=%b Y2=%b Y3=%b", I, S0, S1, Y0, Y1, Y2, Y3);

    I = 1;

    S0 = 0; S1 = 0; #10;
    S0 = 0; S1 = 1; #10;
    S0 = 1; S1 = 0; #10;
    S0 = 1; S1 = 1; #10;

    $finish;

end

endmodule
